Paramètres d'animation

/* Reveal on scroll (hero, sections, cards) */ .reveal { opacity: 0; transform: translateY(18px); filter: blur(4px); /* hero seulement */ transition: all 0.7s cubic-bezier(0.16, 1, 0.3, 1); transition-delay: var(--delay, 0s); } /* Stagger delays: 0s, 0.12s, 0.24s, ... */ /* IntersectionObserver threshold: 0.2 to 0.3 */ /* GSAP counter animation */ duration: 2s ease: 'power2.out' delay per item: 0.15s /* Testimonial columns scroll */ animation: scroll-up 25-30s linear infinite hover: animation-play-state: paused /* Partner logos scroll */ animation: scroll-left 30s linear infinite
